    const label = document.getElementById("triptych-label");

    let spinning = true;

    function getRandomUnique(exclude = []) {

      const available = images.filter(img => !exclude.includes(img));

      return available[Math.floor(Math.random() * available.length)];

    }


    function setRandomTriptych() {

      const chosen = [];

      slots.forEach((slot) => {

        const next = getRandomUnique(chosen);

        chosen.push(next);

        slot.src = next;

      });

    }


    function startSpin() {

      spinning = true;

      label.textContent = "Click to stop";
